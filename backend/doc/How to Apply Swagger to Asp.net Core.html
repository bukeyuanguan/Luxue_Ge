    How to Apply Swagger to Asp.net Core

    Skip to end of metadata
    • Created by Anonymous, last modified by Xiao Lei Song yesterday at 5:33 PM
    Go to start of metadata

    What is Swagger

    When consuming a web API, understanding its various methods can be challenging for a developer.

    Swagger is a language-agnostic specification for describing REST APIs. It's now referred to as OpenAPI

    Swashbuckle.AspNetCore is an open source project for generating Swagger documents for ASP.NET Core Web APIs

    Related Tool

    • https://editor.swagger.io/
    • Swagger Preview in VS Code

    How to Apply Swagger to Asp.net Core


    Install Package

    Tools -> NuGet Package Manager -> Package Manager Console

    OperationFilter

    If we want to add a common input item, we can use OperationFilter. Following code add Version in request header for all interface

    Code
    Swagger
    Code
    Swagger
    VersionFilter.cs
    ?
    public class VersionFilter : IOperationFilter
    {
        public void Apply(OpenApiOperation operation, OperationFilterContext context)
        {
            if (operation.Parameters == null)
            {
                operation.Parameters = new List<OpenApiParameter>();
            }
     
            operation.Parameters.Add(new OpenApiParameter
            {
                Name = "version",
                In = ParameterLocation.Header,
                Description = "Specify API version",
                Required = true,
                Schema = new OpenApiSchema
                {
                    Type = "string",
                    Default = new OpenApiString("1.0.0.0")
                }
            });
        }
    }
    Startup.cs
    ?
    public class Startup
    {
        public void ConfigureServices(IServiceCollection services)
        {
            services.AddSwaggerGen(c=> {
                c.OperationFilter<VersionFilter>();
            });
        }
    }
